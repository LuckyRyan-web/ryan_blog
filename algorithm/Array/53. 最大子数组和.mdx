## 动态规划

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
    let pre = 0
    let sum = nums[0]

    nums.forEach((v) => {
        pre = Math.max(v, v + pre)

        sum = Math.max(pre, sum)
    })

    return sum
}
```
